void func(string &s, int index, vector<string> &ans)

{

    if (index == s.length())

    {

        ans.push_back(s);

        return;

    }

 

    for (int i = index; i < s.length(); i++)

    {

        swap(s[i],s[index]);

        func(s, index + 1 , ans);

        swap(s[i],s[index]);

 

    }
